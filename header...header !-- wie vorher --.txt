<script>
const container = document.getElementById('skinMakerContainer');
const scene = new THREE.Scene();
const camera = new THREE.PerspectiveCamera(75, 1, 0.1, 1000);
const renderer = new THREE.WebGLRenderer({antialias:true, alpha:true});
renderer.setSize(400,400);
container.appendChild(renderer.domElement);

const loader = new MinecraftSkinViewer.SkinViewer(container, {
  width: 400,
  height: 400,
  skin: 'default_skin.png', // Platzhalter
  zoom:1.5,
  backgroundColor: 0x00000000
});

function animate(){
  requestAnimationFrame(animate);
  loader.model.rotation.y += 0.01;
  renderer.render(scene,camera);
}
animate();

// Skin speichern
document.getElementById('saveSkin').addEventListener('click',()=>{
    loader.exportSkin(); // exportiert Skin als PNG zum Download
});
</script>
